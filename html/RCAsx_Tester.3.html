    5 -- Pluggable tester for a RCAsx.
    6 entity RCAsx_Tester is
    7     generic (n: positive range 1 to 32);
    8     port
    9     ( a, b : out std_logic_vector(n-1 downto 0)
   10     ; ci   : out std_logic
   11     ; s    : in  std_logic_vector(n downto 0)
   12     );
   13 end entity;
   14 
   15 architecture Exhaustive of RCAsx_Tester is
   16     -- Input range.
   17     subtype RCAsx_input is integer range (-2 ** (n-1))
   18                                       to ((2 ** (n-1)) - 1);
   19 begin
   20     process
   21     begin
   22         for i in RCAsx_input loop
   23             for j in RCAsx_input loop
   24                 for k in 0 to 1 loop
   25                     a  <= std_logic_vector(to_signed(i, n));
   26                     b  <= std_logic_vector(to_signed(j, n));
   27                     ci <= '0' when k = 0 else '1';
   28 
   29                     wait for 1 ns;
   30                     assert s = std_logic_vector(to_signed(i + j + k, n + 1))
   31                         report "wrong output"
   32                         severity error;
   33                 end loop;
   34             end loop;
   35         end loop;
   36 
   37         wait;
   38     end process;
   39 end architecture;
